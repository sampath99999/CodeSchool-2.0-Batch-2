CREATE TABLE
    userdetails(
        id serial PRIMARY KEY,
        email varchar(255) not null,
        password varchar(255) not null
    );

CREATE table
    tokens(
        id serial PRIMARY KEY,
        token varchar(255) not null,
        user_id int REFERENCES userdetails(id)
    );
